# SESSION MANAGEMENT (HASH + TTL)
HSET session:<quizId>:<studentId> currentQuestion 0 answers "{}" startTime 1712345678000
EXPIRE session:<quizId>:<studentId> 600       # TTL = quiz duration (seconds)
HGETALL session:<quizId>:<studentId>            # Read session
TTL session:<quizId>:<studentId>                # Check remaining time

# SPAM PREVENTION (NX = only set if key doesn't exist)
SET answer:<quizId>:<studentId>:<qNo> "A" NX EX 600

# LIVE LEADERBOARD (Sorted Set)
ZADD leaderboard:<quizId> NX 0 <studentId>      # Add with initial score
ZINCRBY leaderboard:<quizId> 10 <studentId>      # Increment score on correct
ZREVRANGE leaderboard:<quizId> 0 9 WITHSCORES    # Top 10 with scores
ZREVRANK leaderboard:<quizId> <studentId>         # Get rank (0-indexed)

# END OF QUIZ — Delete session keys
DEL session:<quizId>:<studentId>
DEL answer:<quizId>:<studentId>:0
DEL answer:<quizId>:<studentId>:1
